FUNCTION addNewOrder RETURN BOOLEAN IS

  v_OrderId    VARCHAR2(13);
  v_ShowAlert  NUMBER;
  v_company    varchar2(250);
    
  v_cur_itm     VARCHAR2(180);
  v_cur_itmn1   VARCHAR2(180);
  v_cur_itmn2   VARCHAR2(180);
  v_cur_itmn3   VARCHAR2(180);
  v_cur_itmn4   VARCHAR2(180);
  v_cur_itmn5   VARCHAR2(180);
  vv_cur_itm    VARCHAR2(180);
  v_cur_block   VARCHAR2(180)  :=rtrim('TB_DUTY_SCHEDULE_SETUP1');---:System.Cursor_Block; 
  
  v_cnt         number:= 1;
  vv_cnt        number;
  v_primeno     number;
  
  v_user        varchar2(50);
  
BEGIN
  	

IF :SYSTEM.FORM_STATUS = 'CHANGED' THEN
	
	v_ShowAlert := ShowAlert('             Do  You  Want  to  Save  !!!','2');
	
  	IF (v_ShowAlert = ALERT_BUTTON1) THEN 
  	  go_block('TB_DUTY_SCHEDULE_SETUP');
  	  
  	  v_user    := :global.username;
  	  
  	  :TB_DUTY_SCHEDULE_SETUP.PROCESSBY   := v_user;     
      :TB_DUTY_SCHEDULE_SETUP.PROCESSDATE := sysdate;
     
  	  v_company := :TB_DUTY_SCHEDULE_SETUP.company;
   	  
			COMMIT_FORM;
			CLEAR_block;
			go_block('TB_DUTY_SCHEDULE_SETUP1');
			execute_query;
			
			:TB_DUTY_SCHEDULE_SETUP.company  := v_company;
			go_item('TB_DUTY_SCHEDULE_SETUP.company');
		 
   
  else 
 	return false;
  END IF;
  end if;

		  begin
			select count(*) into vv_cnt
			from TB_DUTY_SCHEDULE_SETUP;
			exception
			when others then vv_cnt := 0;
			end;
			
		--for c1rec in (	select count(*)v_cnt
			--              from tb_company_info)
			first_record;
			while (v_cnt <= vv_cnt)
			
			loop  
	
			v_cur_itmn1   :='SECTION_WORKER';
		  v_cur_itmn2   :='INSTARTTIME';
		  v_cur_itmn3   :='LATESTARTTIME';
		  v_cur_itmn4   :='EXITTIME';
		  v_cur_itmn5   :='OTAPPLICABLE';
		 		            
		  v_cur_itm       :=  Get_Block_Property( v_cur_block, first_item );
			vv_cur_itm      :=  Get_Block_Property( v_cur_block, last_ITEM );
		
			v_cur_itm    := v_cur_block||'.'||v_cur_itm;
			v_cur_itmn1  := v_cur_block||'.'||v_cur_itmn1;
			v_cur_itmn2  := v_cur_block||'.'||v_cur_itmn2;
			v_cur_itmn3  := v_cur_block||'.'||v_cur_itmn3;
			v_cur_itmn4  := v_cur_block||'.'||v_cur_itmn4;
			v_cur_itmn5  := v_cur_block||'.'||v_cur_itmn5;
			vv_cur_itm   := v_cur_block||'.'||vv_cur_itm;  
	
	
			v_primeno  := instr(v_cnt/2,'.');
			
			if v_primeno = 0 then
				
			Set_Item_Instance_Property( v_cur_itm, v_cnt, VISUAL_ATTRIBUTE,'CLEAR_TEXT'); 
			Set_Item_Instance_Property( v_cur_itmn1, v_cnt, VISUAL_ATTRIBUTE,'CLEAR_TEXT');
			Set_Item_Instance_Property( v_cur_itmn2, v_cnt, VISUAL_ATTRIBUTE,'CLEAR_TEXT');
			Set_Item_Instance_Property( v_cur_itmn3, v_cnt, VISUAL_ATTRIBUTE,'CLEAR_TEXT');
			Set_Item_Instance_Property( v_cur_itmn4, v_cnt, VISUAL_ATTRIBUTE,'CLEAR_TEXT');
			Set_Item_Instance_Property( v_cur_itmn5, v_cnt, VISUAL_ATTRIBUTE,'CLEAR_TEXT');
			Set_Item_Instance_Property( vv_cur_itm,v_cnt, VISUAL_ATTRIBUTE,'CLEAR_TEXT'); 
			else
			Set_Item_Instance_Property( v_cur_itm, v_cnt, VISUAL_ATTRIBUTE,'VA_COLOUR'); 
			Set_Item_Instance_Property( v_cur_itmn1, v_cnt, VISUAL_ATTRIBUTE,'VA_COLOUR');
			Set_Item_Instance_Property( v_cur_itmn2, v_cnt, VISUAL_ATTRIBUTE,'VA_COLOUR');
			Set_Item_Instance_Property( v_cur_itmn3, v_cnt, VISUAL_ATTRIBUTE,'VA_COLOUR');
			Set_Item_Instance_Property( v_cur_itmn4, v_cnt, VISUAL_ATTRIBUTE,'VA_COLOUR');
			Set_Item_Instance_Property( v_cur_itmn5, v_cnt, VISUAL_ATTRIBUTE,'VA_COLOUR');
			Set_Item_Instance_Property( vv_cur_itm,v_cnt, VISUAL_ATTRIBUTE,'VA_COLOUR'); 
			end if;
  
		  v_cur_itmn1     :='';
		  v_cur_itmn2     :='';
		  v_cur_itmn3     :='';
		  v_cur_itmn4     :='';
		  v_cur_itmn5     :='';
		  
		  v_cnt := v_cnt + 1;
		  next_record; 	
			end loop;
			
			first_record; 
	
	go_block('TB_DUTY_SCHEDULE_SETUP');
  go_item('TB_DUTY_SCHEDULE_SETUP.SECTION_WORKER');
   
exception
	when others then nuLL;
END addNewOrder;

FUNCTION showAlert(message VARCHAR2, button VARCHAR2) RETURN NUMBER IS
 l_show_alert NUMBER;
BEGIN
	IF (button = '1') THEN
      SET_ALERT_PROPERTY('A_MSG',ALERT_MESSAGE_TEXT,message);
		  l_show_alert := SHOW_ALERT('A_MSG'); 
		  
	ELSIF (button = '2') THEN
      SET_ALERT_PROPERTY('ALERT_CONFIRM',ALERT_MESSAGE_TEXT,message);
		  l_show_alert := SHOW_ALERT('ALERT_CONFIRM'); 
	ELSE
	    SET_ALERT_PROPERTY('ALERT_3_BUTTON',ALERT_MESSAGE_TEXT,message);
	    l_show_alert := SHOW_ALERT('ALERT_3_BUTTON'); 
	END IF;

	  RETURN l_show_alert ;
END;
